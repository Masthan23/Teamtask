const SHEET_NAME = "Scrapping Discussion";

function doGet(e) {
  try {
    const action = e.parameter.action;
    const ss = SpreadsheetApp.getActiveSpreadsheet();
    const sheet = ss.getSheetByName(SHEET_NAME);
    if (!sheet) return jsonOut({ success: false, message: "Sheet not found: " + SHEET_NAME });
    if (action === "ping") {
      return jsonOut({ success: true, message: "Connected to " + SHEET_NAME + " (" + (sheet.getLastRow()-1) + " rows)" });
    }
    if (action === "getAll") {
      const lastRow = sheet.getLastRow();
      if (lastRow <= 1) return jsonOut({ success: true, rows: [] });
      const data = sheet.getRange(1, 1, lastRow, 20).getValues();
      function fmtDate(val) {
        if (!val) return "";
        if (val instanceof Date && !isNaN(val)) {
          const d=String(val.getDate()).padStart(2,'0'),m=String(val.getMonth()+1).padStart(2,'0'),y=val.getFullYear();
          if (y<2000||y>2100) return "";
          return d+"/"+m+"/"+y;
        }
        const s=String(val).trim();
        return (s==="0"||s==="NaN")?"":s;
      }
      function clean(val) {
        if (val===null||val===undefined) return "";
        const s=String(val).trim();
        return (s==="0"||s==="false")?"":s;
      }
      const rows=data.slice(1).map((row,i)=>({
        rowNum:clean(row[0])||String(i+1), showName:clean(row[1]),
        status:clean(row[2])||"Pending", taskDoneBy:clean(row[3]),
        sourcePlace:clean(row[4]), scrapedFile:clean(row[5]),
        chineseFile:clean(row[6]), chineseBy:clean(row[7]),
        chineseSrt:clean(row[8]), chineseSrtBy:clean(row[9]),
        englishSrt:clean(row[10]), englishSrtBy:clean(row[11]),
        merging:clean(row[12]), mergingBy:clean(row[13]),
        subtitleDocx:clean(row[14]), subtitleDocxBy:clean(row[15]),
        toolScript:clean(row[16]), toolScriptBy:clean(row[17]),
        assignDate:fmtDate(row[18]), completedDate:fmtDate(row[19]),
        _sheetRowIndex:i+2
      })).filter(r=>r.showName!=="");
      return jsonOut({ success:true, rows:rows, total:rows.length });
    }
    return jsonOut({ success:false, message:"Unknown action: "+action });
  } catch(err) { return jsonOut({ success:false, message:"Error: "+err.toString() }); }
}

function doPost(e) {
  try {
    const data=JSON.parse(e.postData.contents), action=data.action;
    const ss=SpreadsheetApp.getActiveSpreadsheet();
    const sheet=ss.getSheetByName(SHEET_NAME);
    if (!sheet) return jsonOut({ success:false, message:"Sheet not found" });
    if (action==="addRow") {
      sheet.appendRow([sheet.getLastRow(),data.showName||"",data.status||"Pending",data.taskDoneBy||"",
        data.sourcePlace||"",data.scrapedFile||"",data.chineseFile||"",data.chineseBy||"",
        data.chineseSrt||"",data.chineseSrtBy||"",data.englishSrt||"",data.englishSrtBy||"",
        data.merging||"",data.mergingBy||"",data.subtitleDocx||"",data.subtitleDocxBy||"",
        data.toolScript||"",data.toolScriptBy||"",data.assignDate||"",data.completedDate||""]);
      SpreadsheetApp.flush();
      return jsonOut({ success:true, message:"Added" });
    }
    if (action==="updateRow") {
      const allData=sheet.getDataRange().getValues(), idx=parseInt(data.sheetRowIndex);
      const makeRow=(orig)=>[orig,data.showName||"",data.status||"Pending",data.taskDoneBy||"",
        data.sourcePlace||"",data.scrapedFile||"",data.chineseFile||"",data.chineseBy||"",
        data.chineseSrt||"",data.chineseSrtBy||"",data.englishSrt||"",data.englishSrtBy||"",
        data.merging||"",data.mergingBy||"",data.subtitleDocx||"",data.subtitleDocxBy||"",
        data.toolScript||"",data.toolScriptBy||"",data.assignDate||"",data.completedDate||""];
      if (idx>=2&&idx<=allData.length) {
        const rn=String(allData[idx-1][1]).trim().toLowerCase(),dn=String(data.oldShowName||data.showName||"").trim().toLowerCase();
        if (rn===dn||rn===String(data.showName||"").trim().toLowerCase()) {
          sheet.getRange(idx,1,1,20).setValues([makeRow(allData[idx-1][0])]);
          SpreadsheetApp.flush(); return jsonOut({ success:true, message:"Updated row "+idx });
        }
      }
      for (let i=1;i=2&&idx<=allData.length) {
        if (String(allData[idx-1][1]).trim().toLowerCase()===String(data.showName||"").trim().toLowerCase()) {
          sheet.deleteRow(idx); SpreadsheetApp.flush(); return jsonOut({ success:true, message:"Deleted row "+idx });
        }
      }
      for (let i=allData.length-1;i>=1;i--) {
        if (String(allData[i][1]).trim().toLowerCase()===String(data.showName||"").trim().toLowerCase()) {
          sheet.deleteRow(i+1); SpreadsheetApp.flush(); return jsonOut({ success:true, message:"Deleted row "+(i+1) });
        }
      }
      return jsonOut({ success:false, message:"Not found: "+data.showName });
    }
    return jsonOut({ success:false, message:"Unknown action: "+action });
  } catch(err) { return jsonOut({ success:false, message:"Error: "+err.toString() }); }
}

function jsonOut(obj) {
  return ContentService.createTextOutput(JSON.stringify(obj)).setMimeType(ContentService.MimeType.JSON);
}
